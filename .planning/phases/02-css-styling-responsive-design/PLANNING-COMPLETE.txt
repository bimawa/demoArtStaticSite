================================================================================
KEY DESIGN DECISIONS
================================================================================

Color Palette:
  Background: #FDFBF7 (warm cream)
  Text: #1A1A1A (soft black)
  Borders: #E8E6E1 (light cream)
  Contrast: 18:1 (exceeds WCAG AA)

Typography:
  Headings: Lora (serif) from Google Fonts
  Body: Nunito (sans-serif) from Google Fonts
  Fallbacks: System fonts if Google Fonts fail
  Font Loading: display=swap to prevent FOIT

Spacing System:
  Base Unit: 8px
  Scale: 8, 16, 24, 32, 48, 64px
  Mobile: 32px section padding
  Tablet: 48px section padding
  Desktop: 64px section padding

Responsive Breakpoints:
  Mobile: < 768px (1 column gallery)
  Tablet: 768px - 1024px (2 column gallery)
  Desktop: > 1024px (3 column gallery)

Hover Effects:
  Transform: translateY(-4px)
  Shadow: 0 8px 16px rgba(26, 26, 26, 0.12)
  Duration: 0.2s ease
  Mobile: Active state on tap
